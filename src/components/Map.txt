import React, { useEffect, useState, useRef, useContext } from "react";
import {
  GoogleMap,
  Marker,
  useJsApiLoader,
  HeatmapLayer,
} from "@react-google-maps/api";
import axios from "axios";
import Menu from "./Menu";
import AuthContext from "../context/AuthContext";
import { useNavigate } from "react-router-dom";
import Table from "react-bootstrap/Table";
import Blue from "../assets/images/blue_ping.png";
import Red from "../assets/images/red_ping.png";

const libraries = ["visualization", "marker"];

const Map = () => {
  const navigate = useNavigate();
  const { brand } = useContext(AuthContext);
  const { isLoaded } = useJsApiLoader({
    googleMapsApiKey: "AIzaSyCyRs-MwlVW94IPBR0BPwkhwhRZX68KoAg", // ← APIキーを設定
    libraries,
  });

  const [markers, setMarkers] = useState([]);
  const [originalMarkers, setOriginalMMarkers] = useState([]);
  const [medium, setMedium] = useState([]);
  const [zoom, setZoom] = useState(10);
  const [iconSize, setIconSize] = useState(null);
  const mapRef = useRef(null);
  const [targetMedium, setTargetMedium] = useState("");
  const [targetBrand, setTargetBrand] = useState("");
  const [targetStatus, setTargetStatus] = useState("");
  const [targetShop, setTargetShop] = useState("");
  const [shop, setShop] = useState([]);
  const initialCenter = { lat: 31.584172816548488, lng: 130.7938207962173 };
  const [monthArray, setMonthArray] = useState([]);
  const [startMonth, setStartMonth] = useState("");
  const [endMonth, setEndMonth] = useState("");
  const [areaList, setAreaList] = useState([]);
  const [filteredAreaList, setFilteredAreaList] = useState([]);
  const [slice, setSlice] = useState(20);
  const [locations, setLocations] = useState([]);
  const [heatmap, setHeatmap] = useState(null);
  useEffect(() => {
    // if (!brand || brand.trim() === "") navigate("/");
    if (!isLoaded) return; // ← ロード完了後のみ実行

    const getYearMonthArray = (startYear, startMonth) => {
      const now = new Date();
      const currentYear = now.getFullYear();
      const currentMonth = now.getMonth() + 1;
      const yearMonthArray = [];
      let year = startYear;
      let month = startMonth;

      while (
        year < currentYear ||
        (year === currentYear && month <= currentMonth)
      ) {
        const formattedMonth = month.toString().padStart(2, "0");
        yearMonthArray.push(`${year}/${formattedMonth}`);
        month++;
        if (month > 12) {
          month = 1;
          year++;
        }
      }
      return yearMonthArray;
    };

    setMonthArray(getYearMonthArray(2025, 1));

    const fetchData = async () => {
      try {
        const headers = {
          Authorization: "4081Kokubu",
          "Content-Type": "application/json",
        };
        const [customerResponse, mediumResponse, shopResponse] =
          await Promise.all([
            axios.post(
              "https://khg-marketing.info/dashboard/api/",
              { demand: "customer_map" },
              { headers }
            ),
            axios.post(
              "https://khg-marketing.info/dashboard/api/",
              { demand: "medium_list" },
              { headers }
            ),
            axios.post(
              "https://khg-marketing.info/dashboard/api/",
              { demand: "shop_list" },
              { headers }
            ),
          ]);
        const filtered = customerResponse.data.filter((item) => item.lat_lng);
        setOriginalMMarkers(filtered);
        setMedium(mediumResponse.data);
        setShop(shopResponse.data);
        const filteredHeatmapData = customerResponse.data.filter(
          (item) => item.lat_lng
        );
        setLocations(
          filteredHeatmapData.map((item) => {
            const [lat, lng] = item.lat_lng.split(",");
            return { lat: parseFloat(lat), lng: parseFloat(lng) };
          })
        );
      } catch (e) {
        console.error("データ取得エラー:", e);
      }
    };

    fetchData();
  }, [isLoaded]);

  // フィルタリング
  useEffect(() => {
    if (!isLoaded) return;

    let startDate;
    if (startMonth !== "") startDate = new Date(`${startMonth}/01`);

    let endDate;
    if (endMonth !== "") {
      const [year, month] = endMonth.split("/").map(Number);
      endDate = new Date(year, month, 0);
    }

    const filtered = originalMarkers.filter((item) => {
      const targetDate = new Date(item.register);
      return (
        (startMonth === "" ? true : targetDate >= startDate) &&
        (endMonth === "" ? true : targetDate <= endDate) &&
        (targetMedium === "" ? true : item.medium === targetMedium) &&
        (targetBrand === "" ? true : item.shop.slice(0, 2) === targetBrand) &&
        (targetStatus === "" ? true : item.status === targetStatus) &&
        (targetShop === "" ? true : item.shop === targetShop)
      );
    });
    setMarkers(filtered);

    const filteredArea = filtered.map((item) => {
      const parts = item.full_address.split(" ");
      const match = parts.slice(0, 3).join("");
      const regex = /[0-9０-９]/;
      const idx = match.search(regex);
      return idx === -1 ? match : match.substring(0, idx);
    });

    const areaList = [...new Set(filteredArea)]
      .map((name) => ({
        name,
        count: filteredArea.filter((area) => area === name).length,
      }))
      .sort((a, b) => b.count - a.count);

    setAreaList(areaList);

    const heatmapData = locations.map(
      (loc) => new window.google.maps.LatLng(loc.lat, loc.lng)
    );

    if (!heatmap) {
      const heatmapLayer = new window.google.maps.visualization.HeatmapLayer({
        data: heatmapData,
      });
      heatmapLayer.setMap(mapRef.current); // ← mapRef.current を使う
      setHeatmap(heatmapLayer);
      console.log(heatmapLayer);
    } else {
      heatmap.setData(heatmapData);
    }
  }, [
    isLoaded,
    startMonth,
    endMonth,
    originalMarkers,
    targetMedium,
    targetBrand,
    targetStatus,
    targetShop,
    locations,
  ]);

  // ズームに応じてマーカーサイズを変更
  useEffect(() => {
    if (!isLoaded) return;

    const baseSize = 16;
    const maxSize = 40;
    const minZoom = 8;
    const maxZoom = 16;

    const clampedZoom = Math.min(Math.max(zoom, minZoom), maxZoom);
    const scale = (clampedZoom - minZoom) / (maxZoom - minZoom);
    const sizeValue = baseSize + scale * (maxSize - baseSize);

    const size = new window.google.maps.Size(sizeValue, sizeValue);
    setIconSize(size);
  }, [isLoaded, zoom]);

  const containerStyle = { width: "100%", height: "700px" };

  if (!isLoaded) {
    return <div>Loading...</div>;
  }
  return (
    <>
      <Menu brand={brand} />
      <div className="d-flex">
        <div className="container my-3">
          {/* フィルタUI */}
          <select
            style={{
              border: "1px solid #D3D3D3",
              fontSize: "13px",
              height: "40px",
              borderRadius: "7px",
              width: "200px",
              marginLeft: "10px",
              paddingLeft: "8px",
            }}
            onChange={(e) => setStartMonth(e.target.value)}
          >
            <option value="" selected>
              開始月
            </option>
            {monthArray.map((month, index) => (
              <option key={index} value={month}>
                {month}
              </option>
            ))}
          </select>
          <span style={{ marginLeft: "3px" }}>～</span>
          <select
            style={{
              border: "1px solid #D3D3D3",
              fontSize: "13px",
              height: "40px",
              borderRadius: "7px",
              width: "200px",
              marginLeft: "10px",
              paddingLeft: "8px",
            }}
            onChange={(e) => setEndMonth(e.target.value)}
          >
            <option value="" selected>
              終了月
            </option>
            {monthArray.map((month, index) => (
              <option key={index} value={month}>
                {month}
              </option>
            ))}
          </select>
          <select
            style={{
              border: "1px solid #D3D3D3",
              fontSize: "13px",
              height: "40px",
              borderRadius: "7px",
              width: "200px",
              marginLeft: "10px",
              paddingLeft: "8px",
            }}
            onChange={(e) => setTargetMedium(e.target.value)}
          >
            <option value="">販促媒体を選択</option>
            {medium.map((item, index) => (
              <option key={index} value={item.medium}>
                {item.medium}
              </option>
            ))}
          </select>
          <select
            style={{
              border: "1px solid #D3D3D3",
              fontSize: "13px",
              height: "40px",
              borderRadius: "7px",
              width: "200px",
              marginLeft: "10px",
              paddingLeft: "8px",
            }}
            onChange={(e) => setTargetStatus(e.target.value)}
          >
            <option value="">ステータスを選択</option>
            <option value="見込み">見込み</option>
            <option value="契約済み">契約済み</option>
            <option value="会社管理">会社管理</option>
            <option value="失注">失注</option>
          </select>
          <select
            style={{
              border: "1px solid #D3D3D3",
              fontSize: "13px",
              height: "40px",
              borderRadius: "7px",
              width: "200px",
              marginLeft: "10px",
              paddingLeft: "8px",
            }}
            onChange={(e) => setTargetBrand(e.target.value)}
          >
            <option value="">ブランドを選択</option>
            <option value="KH">KH</option>
            <option value="DJ">DJH</option>
            <option value="なご">なごみ</option>
            <option value="2L">2L</option>
            <option value="PG">PG HOUSE</option>
            <option value="JH">JH</option>
          </select>
          <select
            style={{
              border: "1px solid #D3D3D3",
              fontSize: "13px",
              height: "40px",
              borderRadius: "7px",
              width: "200px",
              marginLeft: "10px",
              paddingLeft: "8px",
            }}
            onChange={(e) => setTargetShop(e.target.value)}
          >
            <option value="">店舗を選択</option>
            {shop
              .filter((item) => !item.shop.includes("未設定"))
              .map((item) => (
                <option key={item.shop} value={item.shop}>
                  {item.shop}
                </option>
              ))}
          </select>
        </div>
      </div>

      <div className="container d-flex pb-5">
        <div style={{ width: "70%" }}>
          {/* 凡例 */}
          <div
            className="d-flex"
            style={{ fontSize: "12px", marginBottom: "10px" }}
          >
            <div style={{ width: "14px" }}>
              <img src={Blue} className="w-100" />
            </div>
            <div style={{ marginLeft: "4px" }}>来場済み（未契約者）</div>
            <div style={{ width: "14px", marginLeft: "20px" }}>
              <img src={Red} className="w-100" />
            </div>
            <div style={{ marginLeft: "4px" }}>契約者</div>
          </div>

          {/* Google Map */}
          <GoogleMap
            mapContainerStyle={containerStyle}
            zoom={zoom}
            onLoad={(map) => {
              mapRef.current = map;
              map.setCenter(initialCenter);
            }}
            onZoomChanged={() => {
              if (mapRef.current) {
                const currentZoom = mapRef.current.getZoom();
                if (currentZoom !== undefined) {
                  setZoom(currentZoom);
                }
              }
            }}
            options={{
              gestureHandling: "greedy",
              scrollwheel: true,
            }}
          >
            {/* ピン表示 */}
            {markers.map((item, index) => {
              const [latStr, lngStr] = item.lat_lng.split(",");
              const lat = Number(latStr);
              const lng = Number(lngStr);

              return (
                <Marker
                  key={index}
                  position={{ lat, lng }}
                  title={item.label}
                  icon={{
                    url: item.contract ? Red : Blue,
                    scaledSize: iconSize,
                  }}
                />
              );
            })}

            {/* ヒートマップ表示 */}
            <HeatmapLayer
              data={locations.map(
                (loc) => new window.google.maps.LatLng(loc.lat, loc.lng)
              )}
              options={{
                radius: 20,
                opacity: 0.6,
              }}
            />
          </GoogleMap>
        </div>

        {/* サイドテーブル */}
        <div style={{ paddingLeft: "10px", fontSize: "13px", width: "27%" }}>
          <Table>
            <tbody>
              {areaList.slice(slice - 20, slice).map((item) => (
                <tr key={item.name}>
                  <td>{item.name}</td>
                  <td>{item.count}</td>
                </tr>
              ))}
            </tbody>
          </Table>
          <div className="d-flex justify-content-around">
            <div
              onClick={() => {
                if (slice === 20) return;
                setSlice(slice - 20);
              }}
              className={slice === 20 ? "transparent" : "hover"}
            >
              前へ
            </div>
            <div
              onClick={() => {
                setSlice(slice + 20);
              }}
              className={slice >= areaList.length ? "transparent" : "hover"}
            >
              次へ
            </div>
          </div>
        </div>
      </div>
    </>
  );
};

export default Map;
